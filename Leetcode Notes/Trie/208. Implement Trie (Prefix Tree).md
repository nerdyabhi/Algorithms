class Trie {
public:
    bool isTerminal; // to check the last character of a string
    Trie* children[26];  // to store all 26 characters

    Trie() {
        // initialise with false and NULL values in constructor
        this->isTerminal = false;
        for (int i = 0; i < 26; i++) {
            children[i] = NULL;
        }
    }

    void insert(string word) {
        // p = *pointer on the object (i'll use this to iterate on the trie & to check my specific conditions )
        Trie* p = this;

        for (auto ch : word) // iterate over the word
        {
            int idx = ch - 'a'; // yk it already
            if (p->children[idx] == NULL)  // if it's NULL , then we have to allocate a new char trie for it  (better watch youtube tutorials)
            {
                p->children[idx] = new Trie(); // Noice
            }

            p = p->children[idx]; // same as i++ on trie;
        }

        p->isTerminal =true; // set last character's flag to  true (We need it later )
    }

    bool search(string word) {
        Trie* p = this;

        for (auto ch : word) {
            int idx = ch - 'a';
            if (p->children[idx] == NULL)
                return false; // if character isn't present already // why to look after it simply return false
            p = p->children[idx]; // iterate over the trie (i++ on trie XD )
        }

        return p->isTerminal;  // if(the last character is last marked on trie return true; else false  )
    }

    bool startsWith(string prefix) {
        Trie* p = this;

        for (auto ch : prefix) {
            int idx = ch - 'a';
            if (p->children[idx] == NULL) {
                return false;
            }

            p = p->children[idx];
        }

        return true; // we need prefix only , so why to look for it's last or not lol :)
    }
};
