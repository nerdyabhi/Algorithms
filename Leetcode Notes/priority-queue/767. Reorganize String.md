## [767. Reorganize String](https://leetcode.com/problems/reorganize-string/)

```cpp
class Solution {
public:
    string reorganizeString(string str) {
       // we are worried about the most occuring character the most : (greedy greedy)
        int arr[26] = {0}; // get some drip bruh 
        priority_queue<pair<int, char>>pq;

        int n = str.length();

        for(int i=0; i<n; i++) 
        {
            arr[str[i] - 'a']++;
            if(arr[str[i]-'a'] > (n+1)/2) return "";
        }

        for(int i=0; i<26; i++) 
        {
            if(arr[i] >0)
                pq.push({arr[i] , i+'a'}); 
        }

        str = "";

        while(pq.size()>=2) // because we are popping 2 elements at once
        {
            int freq1 = pq.top().first; 
            char ch1 = pq.top().second; 
            pq.pop();
            
            int freq2 = pq.top().first; 
            char ch2 = pq.top().second;
             pq.pop();

            str+=ch1;
            str+= ch2;
            freq1--;
            freq2--;

            if(freq1 > 0) pq.push({freq1 , ch1});
            if(freq2 > 0) pq.push({freq2 , ch2});
            
        }

        if(!pq.empty()) str+=pq.top().second; // incase any element is left in the queue 

        return str;
    }
};
```
