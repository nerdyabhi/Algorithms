## [1405. Longest Happy String](https://leetcode.com/problems/longest-happy-string/description/)

```cpp
class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ans = "";

        priority_queue<pair<int, char>> pq;
        if(a > 0) pq.push({a, 'a'});
        if(b > 0) pq.push({b, 'b'});
        if(c > 0) pq.push({c, 'c'});

        while(pq.size() >= 2) {
            auto p1 = pq.top();
            pq.pop();
            auto p2 = pq.top();
            pq.pop();

            if(p1.first >= 2) {
                ans.append(2, p1.second);
                p1.first -= 2;
            } else {
                ans += p1.second;
                p1.first--;
            }

            if(p1.first < p2.first && p2.first >= 2) { // very very dangerous condition can't understand without dryrun
                ans.append(2, p2.second);
                p2.first -= 2;
            } else {
                ans += p2.second;
                p2.first--;
            }

            if(p1.first > 0) pq.push(p1);
            if(p2.first > 0) pq.push(p2);
        }

        if(!pq.empty()) {
            auto p1 = pq.top();
            pq.pop();
            if(p1.first >= 2) ans.append(2, p1.second);
            else ans += p1.second;
        }

        return ans;
    }
};
```
