## [1962. Remove Stones to Minimize the Total](https://leetcode.com/problems/remove-stones-to-minimize-the-total/)

```cpp
class Solution {
public:
    int minStoneSum(vector<int>& arr, int k) {
     int ans =0;
    priority_queue<int> pq;
       for(int i=0; i<arr.size(); i++) pq.push(arr[i]);

        for(int i=0; i<k; i++)
        {
            int maxi = pq.top();
            pq.pop();
            pq.push(maxi - (maxi/2));
        }

        while(!pq.empty())
        {
           ans+=pq.top(); pq.pop();
        }

        return ans;
    }
};
```
